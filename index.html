// V41 后端 Apps Script const SHEET_ID = "你的 Google Sheet ID"; // 替换为你的 Sheet ID const SHEET_MAP = { maintainTable: "Maintenance", inspectTable: "Inspection" }; function doPost(e) { const data = JSON.parse(e.postData.contents); const action = data.action; const sheet = data.sheet; const id = data.id; const values = data.values; const ss = SpreadsheetApp.openById(SHEET_ID); try { if(action === "login"){ // 简单示例账号验证 const u = values.username; const p = values.password; if(u === "admin" && p === "admin123") return ContentService.createTextOutput(JSON.stringify({success:true,role:"admin"})).setMimeType(ContentService.MimeType.JSON); if(u === "guest" && p === "guest123") return ContentService.createTextOutput(JSON.stringify({success:true,role:"guest"})).setMimeType(ContentService.MimeType.JSON); return ContentService.createTextOutput(JSON.stringify({success:false,error:"账号或密码错误"})).setMimeType(ContentService.MimeType.JSON); } if(!sheet || !SHEET_MAP[sheet]) return ContentService.createTextOutput(JSON.stringify({success:false,error:"无效表名"})).setMimeType(ContentService.MimeType.JSON); const sh = ss.getSheetByName(SHEET_MAP[sheet]); const dataRange = sh.getDataRange(); const rows = dataRange.getValues(); if(action === "add" && values){ sh.appendRow(values); return ContentService.createTextOutput(JSON.stringify({success:true})).setMimeType(ContentService.MimeType.JSON); } if(action === "update" && values && id){ for(let i=1;i<=rows.length;i++){ if(rows[i-1][9] === id){ sh.getRange(i,1,1,values.length).setValues([values]); return ContentService.createTextOutput(JSON.stringify({success:true})).setMimeType(ContentService.MimeType.JSON); } } return ContentService.createTextOutput(JSON.stringify({success:false,error:"找不到行"})).setMimeType(ContentService.MimeType.JSON); } if(action === "delete" && id){ for(let i=1;i<=rows.length;i++){ if(rows[i-1][9] === id){ sh.deleteRow(i); return ContentService.createTextOutput(JSON.stringify({success:true})).setMimeType(ContentService.MimeType.JSON); } } return ContentService.createTextOutput(JSON.stringify({success:false,error:"找不到行"})).setMimeType(ContentService.MimeType.JSON); } return ContentService.createTextOutput(JSON.stringify({success:false,error:"未知操作"})).setMimeType(ContentService.MimeType.JSON); } catch(err){ return ContentService.createTextOutput(JSON.stringify({success:false,error:err.message})).setMimeType(ContentService.MimeType.JSON); } }
